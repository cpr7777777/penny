function showQuestion(){
  const area=document.getElementById("question-area");
  area.innerHTML="";
  const q=questions[currentQuestion];
  const p=document.createElement("p");
  p.textContent=q.q;
  p.style.fontSize="18px";
  p.style.fontWeight="bold";
  p.style.color="#fff"; // ensure question is white
  area.appendChild(p);

  if(q.type==="yesno"){
    const yes=document.createElement("button");
    yes.textContent="Yes";
    yes.onclick=()=>{answers.push(2); nextQuestion();}
    area.appendChild(yes);
    const no=document.createElement("button");
    no.textContent="No";
    no.onclick=()=>{answers.push(1); nextQuestion();}
    area.appendChild(no);
  } else if(q.type==="text"){
    const input=document.createElement("input");
    input.type="text";
    input.placeholder="Type your answer here";
    area.appendChild(input);
    const submit=document.createElement("button");
    submit.textContent="Submit";
    submit.onclick=()=>{
      answers.push(input.value.length || 1);
      nextQuestion();
    };
    area.appendChild(submit);
  }
}

function nextQuestion(){
  currentQuestion++;
  if(currentQuestion<questions.length){
    showQuestion();
  } else {
    calculateFortune();
  }
}

function calculateFortune(){
  let total=answers.reduce((a,b)=>a+b,0);

  document.getElementById("fortune-output").textContent=fortunes[index]; // fortune is now white via CSS
  document.getElementById("quiz-container").style.display="none";
  answers=[];
  currentQuestion=0;
}
